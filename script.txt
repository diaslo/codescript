-- Функция для анимации загрузки
local function load()
	main.Visible = false
	local rotationTween = ts:Create(loadimage, TweenInfo.new(3, Enum.EasingStyle.Linear, Enum.EasingDirection.Out), {Rotation = 180})

	-- Запуск анимации вращения
	rotationTween:Play()

	-- Ожидание окончания анимации вращения
	wait(3)

	-- Создание анимации изменения прозрачности для loadimage
	local transparencyTween = ts:Create(loadimage, TweenInfo.new(1, Enum.EasingStyle.Linear, Enum.EasingDirection.Out), {ImageTransparency = 1})

	transparencyTween:Play()

	wait(1)

	main.Visible = true
end

local url = "https://jsonplaceholder.typicode.com/posts/1"

-- Функция, выполняющая HTTP-запрос
local function makeRequest()
	-- Выполняем GET-запрос
	local success, response = pcall(function()
		return HttpService:GetAsync(url)
	end)

	-- Проверяем успешность запроса
	if success then
		print("HTTP запрос выполнен успешно:")
		print(response) -- Выводим результат запроса
	else
		warn("Ошибка HTTP запроса:", response) -- Если есть ошибки, выводим их в консоль
	end
end

-- Вызываем функцию для выполнения запроса
makeRequest()
